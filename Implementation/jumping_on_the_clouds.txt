using namespace std;


int main(){
    int n,step=0;
    cin >> n;
    vector<int> c(n);
    for(int c_i = 0;c_i < n;c_i++){
       cin >> c[c_i];
    }
    for(int i=0;i<n-1;i++){
        if(c[i+2]==0&&n>=i+2){
            step++;
            i++;
        }
        else{
            step++;
        }
    }
    cout<<step;
    return 0;
}
